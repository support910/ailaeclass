import { Step, Steps } from 'fumadocs-ui/components/steps';


<Steps>
<Step>
### Go to ailaeclass

Go to [app.ailaeclass.com](https://app.ailaeclass.com). Click on the "Login" button on the top right to take you to the Login page.

![ailaeclass Page](/ailaeclass.webp)

</Step>

<Step>
### Sign In

Sign in with your email & password.

![ailaeclass Page](/login-page.webp)

</Step>

<Step>
### Go to the Settings page

Click on the "Settings" button on the lower left corner on your dashboard page and you'll be routed to the profile page.

![Unedited Profile Page](/unedited-profile-page.webp)

</Step>

<Step>
### Make Changes

Upload a profile picture and edit any input field you want to then click "Update Profile".

![Edited Profile Page](/edited-profile-page.webp)

</Step>
</Steps>
